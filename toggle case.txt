import java.util.*;
public class Main{
  public static void main(String[] args){
    Scanner sc=new Scanner(System.in);
    String s=sc.next();
    String r="";
    for(int i=0;i<s.length();i++){
      char ch=s.charAt(i);
      if(ch>='A'&&ch<='Z') r+=Character.toLowerCase(ch);
      else r+=Character.toUpperCase(ch);
    }
    System.out.println(r);
  }
}
